push()

Adiciona um ou mais elementos no fim de um array e retorna o comprimento do array. 

  var emp = ['João', 'Maria', 'Augusto'];
  var emp2 = emp.push('3')
  
  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ 'João', 'Maria', 'Augusto', '3' ]
  4

pop()

remove o último elemento de um array e retorna esse elemento.

  var emp = ['João', 'Maria', 'Augusto'];
  var emp2 = emp.pop()
  
  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ 'João', 'Maria' ]
  Augusto

shift()

Remove o primeiro elemento de um array e retorna esse elemento.

  var emp = ['João', 'Maria', 'Augusto'];
  var emp2 = emp.shift()
  
  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ 'Maria', 'Augusto' ]
  João

unshift()

Adiciona um ou mais elementos ao início do array e retorna seu novo comprimento.

  var emp = ['João', 'Maria', 'Augusto'];
  var emp2 = emp.unshift('Bia','Ana')
  
  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ 'Bia','Ana','João', 'Maria', 'Augusto']
  5

splice()

Remove elementos de um array e (opcionalmente) os substitui, e retorna os itens que foram removidos do array

emp.splice(1,3,'a','b','c','d') - primeiro argumento se refere ao indice do array e o segundo se refere a quantidade de itens que srá retirada a partir do indice indicado no primeiro argumento. os demais ítens são os que serão acrescentados no novo array

  var emp = ['1', '2', '3', '4', '5'];
  var emp2 = emp.splice(1,3,'a','b','c','d') 

  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ '1', 'a', 'b', 'c', 'd', '5' ]
  [ '2', '3', '4' ]

sort()

Ordena os elementos de um array.

  var emp = ['João', 'Maria', 'Augusto'];
  var emp2 = emp.sort()
  
  console.log(emp)
  console.log(emp2)
  --------------------------------------------
  [ 'Augusto', 'João', 'Maria' ]
  [ 'Augusto', 'João', 'Maria' ]

indexOf()

Busca um elemento no array e retorna o índice da primeira ocorrência.

  var emp = ['a', 'b', 'a', 'c', 'd', 'b', 'c', 'b'];

  console.log(emp.indexOf('b'))
  console.log(emp.indexOf('b',3))
  console.log(emp.indexOf('z'))
  --------------------------------------------
  1
  5
  -1
